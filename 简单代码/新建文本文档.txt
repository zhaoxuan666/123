											s = -1;
										int M=(str[0]=='-' || str[0] == '+') ? 1 : 0;
					for(int i =M;  i < n; ++i){
					if(!( str[i] <= '9'&&'0' <= str[i]) ) 
																	return 0;
			res =res=res*10+str[i]-'0';
						}?

	}
};
